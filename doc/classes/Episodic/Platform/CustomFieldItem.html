      def values
        
        if (@values.nil?)
          @values = []
          _values = @item["value"]  
          unless (_values.is_a?(Array))
            _values = [_values]
          end
          _values.compact.each do |v|
            if (v.is_a?(Hash))
              @values << [v["id"].strip, v["content"].strip]
            else
              @values << v.strip
            end
          end
        end
        
        return @values
      end
