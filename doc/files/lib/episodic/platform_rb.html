def require_library_or_gem(library, gem_name = nil)
  if RUBY_VERSION >= '1.9'
    gem(gem_name || library, '>=0') 
  end
  require library
rescue LoadError => library_not_installed
  begin
    require 'rubygems'
    require library
  rescue LoadError
    raise library_not_installed
  end
end
